- 주어진 문자 배열에서 중복되지 않는 문자만을 뽑아내 배열로 반환해주는 함수 ex) ["a", "b", "c", "a", "e", "d", "c"]  ->  ["b", "e" ,"d"]


		func arr(sum:Character...){
	    var resultArray: [Character] = []
	    for i in sum {
	        if sum.filter({ $0 == i }).count == 1 {
	            resultArray.append(i)
	        }
	    }
	    print(resultArray)
	}
