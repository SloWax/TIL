- 자연수를 입력받아 그 수의 약수를 모두 출력하는 함수

func nalNum(x: Int) -> [Int] {
    var arr = [Int]()
    for y in 1 ... x{
        if x % y == 0{
            arr.append(y)
        }
    }
    return(arr)
}
